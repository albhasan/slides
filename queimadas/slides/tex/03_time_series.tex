\begin{frame}
  \frametitle{Time series' components}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item Structural time series model ~\cite{harvey1990}:
        \begin{itemize}
          \item Trend $g(t)$.
          \item Periodic changes $s(t)$.
          \item Effects of holidays $h(t)$.
        \end{itemize}
        \item \textbf{Assumptions}:
        \begin{itemize}
          \item $\epsilon_{t}$ is normally distributed.
          \item Forecasting is a curve-fitting problem; no account for temporal
            dependence structure in the data~\cite{taylor2018}.
        \end{itemize}
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{math}
        \centering
        y(t) = g(t) + s(t) + h(t) + \epsilon_{t}
      \end{math}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{Do the models fit?}
  \begin{itemize}
    \item Do the data have a trend when smoothed?
    \item Do the data have a trend after applying the logarithm function and smoothing?
    \item Are the data autocorrelated in the time dimension?
    \item Do the data have a trend component?
    \item Do the data have season component?
    \item How does the model fit its observations?
    \item How do compare the residuals and the fitted values?
    \item Do the mode-observations residual have a normal distribution?
  \end{itemize}
\end{frame}
